import { useState } from 'react';
import { useRouter } from 'next/router';
import Layout from '../components/Layout';
import Hero from '../components/ui/Hero';
import GenreBar from '../components/ui/GenreBar';
import ContentRow from '../components/ui/ContentRow';
import SearchBar from '../components/ui/SearchBar';
import SearchResults from '../components/ui/SearchResults';
import DetailModal from '../components/ui/DetailModal';
import { ANIME, ROWS } from '../lib/animeData';

export default function Discover() {
  const router = useRouter();
  const [search, setSearch] = useState('');
  const [genre, setGenre] = useState('All');
  const [selected, setSelected] = useState(null);

  const results = search
    ? ANIME.filter(a => a.title.toLowerCase().includes(search.toLowerCase()))
    : null;

  const goToJournal = () => router.push('/journal');

  return (
    <Layout>
      <div style={{ paddingBottom: 60 }}>
        <SearchBar search={search} setSearch={setSearch} />

        {search ? (
          <SearchResults results={results} search={search} onSelect={setSelected} onAddJournal={goToJournal} />
        ) : (
          <>
            <Hero anime={ANIME[0]} onSelect={setSelected} onAddJournal={goToJournal} />
            <GenreBar genre={genre} setGenre={setGenre} />
            {ROWS.map(row => (
              <ContentRow key={row.label} label={row.label} anime={row.anime} progress={row.progress} onSelect={setSelected} />
            ))}
          </>
        )}

        {selected && <DetailModal anime={selected} onClose={() => setSelected(null)} onAddJournal={goToJournal} />}
      </div>
    </Layout>
  );
}
